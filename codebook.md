 	Activity type: value of the following
 	WALKING
 	WALKING_UPSTAIRS
 	WALKING_DOWNSTAIRS
 	SITTING
 	STANDING
 	LAYING


 "subject" 2
 
 	ID of subject that performed the activity
 	1..30
